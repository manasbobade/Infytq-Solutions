def max_frequency_word_counter(data):
    word=""
    frequency=0
    frequency_counter={}
    data=data.lower()
    wordlist=data.split()
    max=1
    maxwordlist=[]
    for i in wordlist:
        if i not in frequency_counter:
            frequency_counter[i]=1
        else:
            frequency_counter[i]+=1
            if (frequency_counter[i]>max):
                max=frequency_counter[i]
    for key,value in frequency_counter.items():
        if value==max:
            if(len(key)>len(word)):
                word=key
            frequency=value


    print(word,frequency)
