def sms_encoding(data):
    #start writing your code here
    wordlist=data.split()
    vowel="aeiouAEIOU"
    returnlist=[]
    condition=True
    for i in wordlist:
        str=""
        for j in i:
            if j in vowel:
                condition=True
            else:
                condition=False
                break
        if(condition==True):
            returnlist.append(i)
        else:
            for k in i:
                if k not in vowel:
                    str+=k
            returnlist.append(str)
    return " ".join(returnlist)



data="I love Python"
print(sms_encoding(data))
